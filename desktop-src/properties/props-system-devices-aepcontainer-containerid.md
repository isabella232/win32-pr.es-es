```
propertyDescription
   name = System.Devices.AepContainer.ContainerId
   shellPKey = PKEY_Devices_AepContainer_ContainerId
   formatID = 0BBA1EDE-7566-4F47-90EC-25FC567CED2A
   propID = 12
   SearchInfo
      InInvertedIndex = false
      IsColumn = false
   typeInfo
      type = Guid
```
